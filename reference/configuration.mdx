## Overview

The `cypress.config.js` file contains all configuration options for the Cypress E2E automation framework. This reference documents every option available in the configuration.

## Base Configuration

```javascript
const { defineConfig } = require("cypress");
const webpackPreprocessor = require('@cypress/webpack-preprocessor');

module.exports = defineConfig({
  // Configuration options
});
```

## Reporter Options

<Accordion title="cypress-mochawesome-reporter">

<ParamField path="reporter" type="string" default="cypress-mochawesome-reporter">

<ParamField path="reporterOptions.reportDir" type="string" default="evidencia/reporte">

</ParamField>

<ParamField path="reporterOptions.charts" type="boolean" default={true}>
  Enable charts and graphs in the test report
</ParamField>

<ParamField path="reporterOptions.reportPageTitle" type="string" default="Reporte de Pruebas">
  Title displayed on the test report page
</ParamField>

<ParamField path="reporterOptions.embeddedScreenshots" type="boolean" default={true}>

</ParamField>

<ParamField path="reporterOptions.inlineAssets" type="boolean" default={true}>
  Inline all assets (CSS, JS, images) into a single HTML file for easy sharing
</ParamField>

### Node Events Setup

The `setupNodeEvents` function configures browser launch options, file preprocessors, and environment variables.

#### Browser Launch Configuration

```javascript
on('before:browser:launch', (browser = {}, launchOptions) => {
  if (browser.family === 'chromium' && browser.name !== 'electron') {
    launchOptions.args.push('--disable-dev-shm-usage');
    launchOptions.args.push('--disable-gpu');
  }
  return launchOptions;
});
```

**Chrome Launch Arguments:**
- `--disable-dev-shm-usage`: Prevents shared memory issues in Docker/CI environments
- `--disable-gpu`: Disables GPU hardware acceleration

#### File Preprocessor

```javascript
const options = {
  webpackOptions: require('./webpack.config.js'),
  watchOptions: {},
};
on('file:preprocessor', webpackPreprocessor(options));
```

Configures Webpack for processing test files, enabling modern JavaScript features and imports.

#### Environment Variables

Environment variables are loaded from system environment and made available to tests:

```javascript
config.env.SECRET_VALUE = process.env.CYPRESS_SECRET_VALUE || config.env.SECRET_VALUE;
config.env.VARIABLE_REPO = process.env.CYPRESS_VARIABLE_REPO || config.env.VARIABLE_REPO;
```

**Available Environment Variables:**
- `CYPRESS_SECRET_VALUE`: Secret value from environment or config
- `CYPRESS_VARIABLE_REPO`: Repository variable from environment or config

</Accordion>

## Example Configuration

```javascript
const { defineConfig } = require("cypress");
const webpackPreprocessor = require('@cypress/webpack-preprocessor');

module.exports = defineConfig({
  reporter: 'cypress-mochawesome-reporter',
  reporterOptions: {
    reportDir: 'evidencia/reporte', 
    charts: true,
    reportPageTitle: 'Reporte de Pruebas',
    embeddedScreenshots: true,
    inlineAssets: true,
  },
  retries: {
    runMode: 2, 
    openMode: 0 
  },
  e2e: {
    screenshotsFolder: 'evidencia/screenshots',
    setupNodeEvents(on, config) {
      on('before:browser:launch', (browser = {}, launchOptions) => {
        if (browser.family === 'chromium' && browser.name !== 'electron') {
          launchOptions.args.push('--disable-dev-shm-usage');
          launchOptions.args.push('--disable-gpu');
        }
        return launchOptions;
      });
      
      const options = {
        webpackOptions: require('./webpack.config.js'),
        watchOptions: {},
      };
      on('file:preprocessor', webpackPreprocessor(options));
      require('cypress-mochawesome-reporter/plugin')(on);
      
      config.env.SECRET_VALUE = process.env.CYPRESS_SECRET_VALUE || config.env.SECRET_VALUE;
      config.env.VARIABLE_REPO = process.env.CYPRESS_VARIABLE_REPO || config.env.VARIABLE_REPO;
      
      return config;
    },
    baseUrl: 'https://www.leagueoflegends.com/',
    viewportWidth: 1920,
    viewportHeight: 1080,
    defaultCommandTimeout: 18000,
    chromeWebSecurity: false
  },
});
```
